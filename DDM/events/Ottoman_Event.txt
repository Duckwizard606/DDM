country_event = {

	id = 90025 # Independence for Tunisia

	title = "EVTNAME90025"
	desc = "EVTDESC90025"
	picture = "Ottoman"

	trigger = {
		tag = TUR
		war = no
		TUN = { vassal_of = TUR }
		OR = {
			year = 1881
			has_country_flag = ottoman_western_democracy
			AND = {
				is_greater_power = no
				is_secondary_power = no
			}
			TUN = { civilized = yes }
		}
	}

	fire_only_once = yes
	mean_time_to_happen = {
		months = 6
		modifier = {
			factor = 10
			NOT = { year = 1881 }
			is_greater_power = no
			is_secondary_power = no
			TUN = { civilized = no }
		}
		modifier = {
			factor = 0.5
			has_global_flag = berlin_congress_held
		}
	}

	option = {
		name = "Let them go"
		badboy = -2
		prestige = -5
		release_vassal = TUN
		relation = {
			who = TUN
			value = 50
		}
		random_owned = {
			limit = {
				owner = { is_greater_power = yes }
			}
			owner = {
				diplomatic_influence = {
					who = TUN
					value = 50
				}
			}
		}
		ai_chance = {
			factor = 60
			modifier = {
				factor = 0.5
				is_greater_power = yes
			}
			modifier = {
				factor = 0
				has_country_flag = ottoman_islamic_caliphate
			}
		}
	}
	option = {
		name = "Tunisia is part of the Empire!"
		release_vassal = TUN
		relation = {
			who = TUN
			value = -100
		}
		random_owned = {
			limit = {
				owner = { is_greater_power = yes }
			}
			owner = {
				diplomatic_influence = {
					who = TUN
					value = -100
				}
			}
		}
		leave_alliance = TUN
		casus_belli = {
			target = TUN
			type = make_puppet
			months = 60
		}
		ai_chance = {
			factor = 40
			modifier = {
				factor = 2
				is_greater_power = yes
			}
		}
	}
}
