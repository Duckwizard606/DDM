country_event = {
	id = 87603
	title = "Jim Crow Laws"
	desc = "With the end of slavery, most of the ex-slave states have enacted segregationist laws."
	picture = "dixie"

		tag = USA
		slavery = no_slavery
		has_global_flag = american_civil_war_has_happened
		any_owned_province = {
			is_core = CSA
		}
		NOT = {
			exists = CSA
			has_country_flag = jim_crow_laws
		}
	}
	
	mean_time_to_happen = {
		months = 120
		modifier = {
			factor = 0.5 #increase likelihood to happen
			has_country_flag = kkk
		}
		modifier = {
			factor = 0.5 #increase likelihood to happen
			year = 1876
		}
	}
	
	option = {
		name = "EVTOPTA16415"
		set_country_flag = jim_crow_laws
		any_owned = {
			limit = { is_core = CSA }
			add_province_modifier = {
				name = jim_crow
				duration = -1
			}
			any_pop = {
				limit = { has_pop_culture = dixie }
				ideology = {
					value = conservative
					factor = 0.40
				}
				ideology = {
					value = reactionary
					factor = 0.20
				}			
				militancy = -2
				consciousness = -4
			}
		}
	}
}

				NOT = { in_sphere = THIS }
				NOT = { is_sphere_leader_of = THIS }

			any_greater_power = {
				alliance_with = THIS
				NOT = { is_our_vassal = THIS }
				NOT = { is_sphere_leader_of = THIS }
			}
		}
	}
	
	mean_time_to_happen = {
		months = 360
	}
	
	option = {
		name = "Ah, blast."

		random_country = {
			limit = {
				alliance_with = THIS
				NOT = { vassal_of = THIS }
				NOT = { is_our_vassal = THIS }
				NOT = { in_sphere = THIS }
				NOT = { is_sphere_leader_of = THIS }
			}
			leave_alliance = THIS
			relation = { who = THIS value = -50 }
		}	
	}
}
