# 880010 - Reactionary Stronghold
province_event = {
	id = 880010
	title = "EVTNAME880010"
	desc = "EVTDESC880010"
	
	trigger = {
		owner = {
			civilized = yes
			num_of_cities = 50
			OR = {
				average_consciousness = 5
				average_militancy = 2
				reactionary = 5
			}
		}
		OR = {
			owner = {
				NOT = {
					any_owned_province = { has_province_modifier = reactionary_stronghold }
				}
			}
			AND = {
				NOT = { has_province_modifier = reactionary_stronghold }
				state_scope = {
					any_owned_province = { has_province_modifier = reactionary_stronghold }
				}
			}
		}
		is_colonial = no
		is_core = THIS
		total_pops = 25000
		reactionary = 15
		NOT = {
			has_province_modifier = reactionary_stronghold
			has_province_modifier = conservative_stronghold
			has_province_modifier = liberal_stronghold
			has_province_modifier = anarcho_liberal_stronghold
			has_province_modifier = socialist_stronghold
			has_province_modifier = communist_stronghold
			has_province_modifier = fascist_stronghold
		}
	}
	
	mean_time_to_happen = {
		months = 400		
		modifier = {
			factor = 0.3
			owner = { has_country_flag = liberal_revolution_in_progress }
		}
		modifier = {
			factor = 0.1
			owner = { has_country_modifier = springtime_of_nations }
		}
		modifier = {
			factor = 0.8
			reactionary = 20
		}
		modifier = {
			factor = 0.8
			reactionary = 25
		}
		modifier = {
			factor = 0.5
			reactionary = 30
		}
		modifier = {
			factor = 0.5
			reactionary = 40
		}
		modifier = {
			factor = 0.5
			reactionary = 50
		}
	}
	
	option = {
		name = "They had best not cause trouble."	
		state_scope = {
			any_owned = {
				limit = {
					NOT = { has_province_modifier = reactionary_stronghold }
				}
				remove_province_modifier = conservative_stronghold
				remove_province_modifier = liberal_stronghold
				remove_province_modifier = anarcho_liberal_stronghold
				remove_province_modifier = socialist_stronghold
				remove_province_modifier = communist_stronghold
				remove_province_modifier = fascist_stronghold
				add_province_modifier = {
					name = reactionary_stronghold
					duration = -1
				}
				any_pop = {
					ideology = {
						value = reactionary
						factor = 0.20
					}
					scaled_consciousness = {
						ideology = reactionary
						factor = -6
					}
				}
			}
		}
	}
}
